import java.util.Arrays;

public class Main {
    public static void main(String[] args){
        String[] names = {"Josue", "Junior", "Nataly", "Dallys", "Yari" }; //Create a new Array
        System.out.println(Arrays.toString(names)); //Print the array
        System.out.println(names.length); //Return the length of the array
        Arrays.sort(names); //Sort
        System.out.println(Arrays.toString(names)); //Order A-Z
    }
}
